Maximum and minimum of an array using minimum number of comparisons

public static int getAns(int arr[]){ 
    int max=arr[0], min=arr[0];
    for(int i = 0; i < arr.length; i++){
        if(arr[i] >max){
            max=arr[i];
        }
        else if(arr[i]<min){
            min=arr[i];
        }
    }
}


   // i + minJump <= j <= min(i + maxJump, s.length - 1), and
   // s[j] == '0'.


public boolean canReach(String s, int minJump, int maxJump) {
    
        int L=s.length();

        int i=0,j=1;


        while (j < L) {
            if(s.charAt(j) == '0' && (i + minJump <= j) && (j <= Math.min(i + maxJump,))){
                i=j;
            }
            j++;
        }
        if(i==L-1){
            return true;
        }
        return false;
}

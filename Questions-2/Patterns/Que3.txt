public class Que3 {

    public static void print(int n) {
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j <= i; j++) {
                System.out.print("*\t");
            }
            System.out.println();
        }

    }

    public static void main(String[] args) {
        print(5);
    }


}
